核心数据结构 库的核心是两种状态结构体：

struct jpeg_decompress_struct：解压缩（读）状态。

struct jpeg_compress_struct：压缩（写）状态。

struct jpeg_error_mgr：错误管理器，必须由用户初始化并绑定到上述结构体，否则错误时将直接 exit()。

所有操作都围绕这两个结构体展开。

错误处理 libjpeg 默认的错误处理方式是调用 exit() 终止程序。为了在函数中安全地捕获错误，必须自定义错误管理器，利用 setjmp/longjmp 实现异常风格的跳转。

libjpeg-turbo

libjpeg 和 libjpeg-turbo 是两个 JPEG 编解码库，功能目标相同但实现细节差异很大。简单说：libjpeg 是原始标准实现，libjpeg-turbo 是保持 API 兼容的高性能分支。

如何选择

  • 追求性能：优先选 libjpeg-turbo，尤其是服务器端处理大量图片、实时压缩等场景。
  • 追求最小依赖或特殊平台：如果目标平台没有 SIMD 支持（如某些嵌入式 CPU），libjpeg 的纯 C 实现更省事。
  • 兼容性顾虑：两者 API 兼容，但 ABI 可能不完全一致（如数据结构对齐），在编译阶段替换即可，不要混用不同库的二进制。

#include <iostream>
#include <cstdio>
#include <csetjmp>   // for setjmp/longjmp
#include <jpeglib.h>

struct error_mgr {
    struct jpeg_error_mgr pub;   /* 公共字段 */
    jmp_buf jmp_buffer;       /* 用于跳转 */
};

// #define METHODDEF(type)          static type
METHODDEF(void) error_exit(j_common_ptr cinfo) {
    const auto err = reinterpret_cast<struct error_mgr *>(cinfo->err);
    /* 可在此打印默认错误信息 */
    (*cinfo->err->output_message)(cinfo);
    /* 跳回调用点 */
    longjmp(err->jmp_buffer, 1);
}

/**
 * 读取 JPEG 文件，返回 RGB24 像素数组。
 * 调用者负责 free() 返回的缓冲区。
 * 失败返回 NULL。
 */
unsigned char* read_jpeg_from_file(const char *filename, unsigned int *width, unsigned int *height) {
    FILE *infile = nullptr;
    if ((infile = fopen(filename, "rb")) == nullptr) {
        fprintf(stderr, "无法打开文件 %s\n", filename);
        return nullptr;
    }

    struct jpeg_decompress_struct cinfo{};
    struct error_mgr err{};
    /* 1. 初始化错误管理器*/
    // libjpeg 默认的错误处理方式是调用 exit() 终止程序。
    // 为了在函数中安全地捕获错误，必须自定义错误管理器，利用 setjmp/longjmp 实现异常风格的跳转。
    cinfo.err = jpeg_std_error(&err.pub);
    err.pub.error_exit = error_exit;

    unsigned char *image_buffer = nullptr;
    // 设置跳跃点, 第一次调用 setjmp 时执行（返回值 0）
    if (setjmp(err.jmp_buffer)) {
        /* 当 longjmp 跳回时执行（返回值 非零）,错误发生时会跳转到这里 */
        jpeg_destroy_decompress(&cinfo);
        fclose(infile);
        if (image_buffer) free(image_buffer);
        return nullptr;
    }

    /* 2. 创建解压缩对象 */
    jpeg_create_decompress(&cinfo);

    /* 3. 指定输入文件 */
    jpeg_stdio_src(&cinfo, infile);

    /* 4. 读取 JPEG 头（必须） */
    // (void) 这个强制类型转换告诉编译器：“我知道这个函数有返回值，但我故意不使用它，不要为此发出警告。
    // jpeg_read_header 的返回值主要标识 JPEG 数据流的大致完整性，但在我们给出的示例中，已经安装了自定义错误管理器（setjmp/longjmp），
    // 任何致命错误都会直接跳转到 setjmp 的错误处理分支，根本不会从函数返回。所以这个返回值在多数安全处理框架下没有意义，直接忽略反而更清晰。
    // 但直接前置 (void) 是最简洁、最常用的“自文档化”手法——它向阅读代码的人表明：此处的返回值是经过考虑的，并非疏忽遗忘。
    (void) jpeg_read_header(&cinfo, TRUE);

    /* 5. 开始解压缩 */
    (void) jpeg_start_decompress(&cinfo);

    *width  = cinfo.output_width;
    *height = cinfo.output_height;
    unsigned int num_channels = cinfo.output_components;  /* 通常为 3 (RGB) 或 1 (灰度) */

    unsigned int row_stride = (*width) * num_channels;

    /* 6. 分配图像缓冲区（存放完整像素） */
    image_buffer = static_cast<unsigned char*>(malloc(row_stride * (*height)));
    if (image_buffer == nullptr) {
        fprintf(stderr, "内存分配失败\n");
        jpeg_destroy_decompress(&cinfo);
        fclose(infile);
        return nullptr;
    }

    /* 7. 逐行读取 */
    uint8_t* buffer_array[1];
    for (unsigned char* ptr = image_buffer;cinfo.output_scanline < cinfo.output_height; ptr += row_stride) {
        buffer_array[0] = ptr;
        (void) jpeg_read_scanlines(&cinfo, buffer_array, 1);
    }

    /* 8. 完成解压缩 */
    (void) jpeg_finish_decompress(&cinfo);

    /* 9. 清理 */
    jpeg_destroy_decompress(&cinfo);
    fclose(infile);

    return image_buffer;
}

/**
 * 如果 JPEG 数据已加载到内存，使用 jpeg_mem_src() 替代 jpeg_stdio_src()
 * 从内存中解压 JPEG，参数 data 为完整 JPEG 数据，size 为其字节数。
 * 返回 RGB 像素缓冲，调用者负责 free()。
 */
unsigned char* read_jpeg_from_memory(const unsigned char *data, unsigned long size, unsigned int *width, unsigned int *height) {
    struct jpeg_decompress_struct cinfo{};
    struct error_mgr err{};
    unsigned char *image_buffer = nullptr;

    cinfo.err = jpeg_std_error(&err.pub);
    err.pub.error_exit = error_exit;
    if (setjmp(err.jmp_buffer)) {
        jpeg_destroy_decompress(&cinfo);
        if (image_buffer) free(image_buffer);
        return nullptr;
    }

    jpeg_create_decompress(&cinfo);

    /* 指定内存源 */
    jpeg_mem_src(&cinfo, data, size);

    (void) jpeg_read_header(&cinfo, TRUE);
    (void) jpeg_start_decompress(&cinfo);

    *width = cinfo.output_width;
    *height = cinfo.output_height;
    unsigned int row_stride = (*width) * cinfo.output_components;

    image_buffer = static_cast<unsigned char*>(malloc(row_stride * (*height)));
    if (!image_buffer) {
        jpeg_destroy_decompress(&cinfo);
        return nullptr;
    }

    /* 7. 逐行读取 */
    uint8_t* buffer_array[1];
    for (unsigned char* ptr = image_buffer;cinfo.output_scanline < cinfo.output_height; ptr += row_stride) {
        buffer_array[0] = ptr;
        (void) jpeg_read_scanlines(&cinfo, buffer_array, 1);
    }

    jpeg_finish_decompress(&cinfo);
    jpeg_destroy_decompress(&cinfo);
    return image_buffer;
}

/**
 * 将 RGB24 像素写入 JPEG 文件。
 * quality 范围 0..100。
 * 成功返回 1，失败返回 0。
 */
int write_jpeg_to_file(const char *filename, const unsigned char *image_buffer, unsigned int width, unsigned int height, int quality) {
    struct jpeg_compress_struct cinfo{};
    struct error_mgr err{};
    FILE *outfile;
    if ((outfile = fopen(filename, "wb")) == nullptr) {
        fprintf(stderr, "无法创建文件 %s\n", filename);
        return 0;
    }

    cinfo.err = jpeg_std_error(&err.pub);
    err.pub.error_exit = error_exit;
    if (setjmp(err.jmp_buffer)) {
        jpeg_destroy_compress(&cinfo);
        fclose(outfile);
        return 0;
    }

    jpeg_create_compress(&cinfo);
    jpeg_stdio_dest(&cinfo, outfile);

    /* 设置图像参数 */
    cinfo.image_width      = width;
    cinfo.image_height     = height;
    cinfo.input_components = 3;          /* RGB = 3 通道 */
    cinfo.in_color_space   = JCS_RGB;    /* 颜色空间 */

    jpeg_set_defaults(&cinfo);
    jpeg_set_quality(&cinfo, quality, TRUE); /* 第二个参数 TRUE 表示基线压缩 */

    jpeg_start_compress(&cinfo, TRUE);

    unsigned int row_stride = width * 3;
    unsigned char* buffer_array[1];
    for (const unsigned char* ptr = image_buffer;cinfo.next_scanline < cinfo.image_height; ptr += row_stride) {
        buffer_array[0] = const_cast<unsigned char*>(ptr);
        (void) jpeg_write_scanlines(&cinfo, buffer_array, 1);
    }

    jpeg_finish_compress(&cinfo);
    jpeg_destroy_compress(&cinfo);
    fclose(outfile);
    return 1;
}

/**
 * 将 RGB 图像压缩到内存，返回 JPEG 数据缓冲区和大小。
 * 调用者必须用 free() 释放返回的缓冲区。
 * 返回 NULL 表示失败。
 */
unsigned char* write_jpeg_to_memory(const unsigned char *image_buffer,
                                    unsigned int width, unsigned int height, int quality,
                                    unsigned long *outsize) {
    struct jpeg_compress_struct cinfo{};
    struct error_mgr err{};
    unsigned char *jpeg_buffer = nullptr;
    *outsize = 0;

    cinfo.err = jpeg_std_error(&err.pub);
    err.pub.error_exit = error_exit;
    if (setjmp(err.jmp_buffer)) {
        jpeg_destroy_compress(&cinfo);
        if (jpeg_buffer) free(jpeg_buffer);
        return nullptr;
    }

    jpeg_create_compress(&cinfo);

    /* 目标为内存 */
    jpeg_mem_dest(&cinfo, &jpeg_buffer, outsize);

    cinfo.image_width = width;
    cinfo.image_height = height;
    cinfo.input_components = 3;
    cinfo.in_color_space = JCS_RGB;

    jpeg_set_defaults(&cinfo);
    jpeg_set_quality(&cinfo, quality, TRUE);

    jpeg_start_compress(&cinfo, TRUE);

    unsigned int row_stride = width * 3;
    unsigned char* buffer_array[1];
    for (const unsigned char* ptr = image_buffer;cinfo.next_scanline < cinfo.image_height; ptr += row_stride) {
        buffer_array[0] = const_cast<unsigned char*>(ptr);
        (void) jpeg_write_scanlines(&cinfo, buffer_array, 1);
    }

    jpeg_finish_compress(&cinfo);
    jpeg_destroy_compress(&cinfo);

    return jpeg_buffer;
}

int main(const int argc, char *argv[]) {
    if (argc < 4) return 1;
    unsigned int width = 0, height = 0;
    const auto rgb_ptr = read_jpeg_from_file(argv[1], &width, &height);
    std::cout << width << " vs " << height << std::endl;

    write_jpeg_to_file(argv[2], rgb_ptr, width, height, 3);

    unsigned long len;
    const auto jpeg_ptr = write_jpeg_to_memory(rgb_ptr, width, height, 3, &len);

    const auto rgb_ptr_2 = read_jpeg_from_memory(jpeg_ptr, len, &width, &height);
    std::cout << width << " vs " << height << std::endl;

    write_jpeg_to_file(argv[3], rgb_ptr_2, width, height, 3);
    return 0;
}
