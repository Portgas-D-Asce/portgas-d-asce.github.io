// (p1, p2) x (p1, p3)
template<typename T>
T cross(const pair<T, T> &p1, const pair<T, T> &p2, const pair<T, T> &p3) {
    return (p2.first - p1.first) * (p3.second - p1.second) -
           (p3.first - p1.first) * (p2.second - p1.second);
}
