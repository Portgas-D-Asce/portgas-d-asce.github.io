template<typename T>
class FenwickTree {
public:
    FenwickTree(int n) : _ft(std::vector<T>(n + 1, 0)) {}
    
    // 0-indexed nums[idx] += x
    void add(int idx, const T& x) {
        idx++;
        while(idx < _ft.size()) {
            _ft[idx] += x;
            idx += lowbit(idx);
        }
    }
    
    // 0-indexed prefix sum [0, idx]
    T query(int idx) const {
        idx++;
        T res = 0;
        while(idx) {
            res += _ft[idx];
            idx -= lowbit(idx);
        }
        return res;
    }
    
    // 0-indexed range sum [p, r]
    T query(int p, int r) const {
        return prefix(++r) - prefix(p);
    }
private:
    static int lowbit(int x) { return x & (-x); }
private:
    vector<T> _ft;
};

template<template<typename> class Op, typename T>
class FenwickTree {
public:
    explicit FenwickTree(const int n, const T& init_value = identity_value<Op, T>) : ori(n, init_value), ft(n + 1, identity_value<Op, T>) {
        linear_init();
    }
    
    explicit FenwickTree(const std::vector<T>& nums) : ori(nums), ft(nums.size() + 1, identity_value<Op, T>) {
        linear_init();
    }
    
    // 0-indexed
    const T& operator[](size_t idx) const {
        int n = ori.size();
        if (idx < 0 || idx >= n) {
            throw std::out_of_range("index out of range");
        }
        return ori[idx];
    }

    bool update(int idx, T x) {
        int n = ori.size();
        if(idx < 0 || idx >= n) {
            throw std::out_of_range("index out of range");
        }
        
        ori[idx] = op(ori[idx], x);
        idx++;
        while(idx < ft.size()) {
            ft[idx] = op(ft[idx], x);
            idx += lowbit(idx);
        }
        return true;
    }

    // 0-indexed [0, idx]
    T query(int idx) const {
        // change to 1-indexed
        idx++;
        T res = identity_value<Op, T>;
        // special scene should return identity
        if(idx < 1) return res;
        // avoid Out-of-bounds access
        if(idx > ori.size()) idx = ori.size();
        while(idx) {
            res = op(res, ft[idx]);
            idx -= lowbit(idx);
        }
        return res;
    }

    // 0-indexed [p, r]
    T query(int p, int r) const {
        // change to 1-indexed
        p++, r++;
        int n = static_cast<int>(ori.size());
        T res = identity_value<Op, T>;
        // special scene should return identity
        if(p > r || r < 1 || p > n) return res;
        
        // void Out-of-bounds access
        if(p < 1) p = 1;
        if(r > n) r = n;
        while(p <= r) {
            if(p <= r - lowbit(r)) {
                res = op(res, ft[r]);
                r -= lowbit(r);
            } else {
                res = op(res, ori[--r]);
            }
        }
        return res;
    }

private:
    static int lowbit(const int x) { return x & (-x); }
    void linear_init() {
        int n = ori.size();
        for(int i = 1; i <= n; ++i) {
            ft[i] = op(ft[i], ori[i - 1]);
            int idx = i + lowbit(i);
            if(idx <= n) ft[idx] = op(ft[idx], ft[i]);
        }
    }
    std::vector<T> ori;
    std::vector<T> ft;
    Op<T> op;
};
