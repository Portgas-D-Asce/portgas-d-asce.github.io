窗口 == 子数组

窗口通常维护了某个性质，在滑动过程中，这个性质具有单调性，需要维护这个性质满足某个条件

最小窗口问题可以转化为 00001111 型二分查找，最大窗口问题可以转化成 11110000 型二分查找，但效率低

first version

最大窗口

// 窗口初始位置及长度
int idx = -1, len = 0;
for(int i = 0, j = 0; i < n; ++i) {
    // 将第 i 个元素添加到窗口里面
    c[s[i] - 'A']++;
    
    // 添加新元素到窗口后可能会使窗口条件遭到破坏
    while(!check(c)){
        // 不断地从窗口左侧移出元素，直至窗口条件重新成立
        c[s[j++] - 'A']--;
    }
    
    // 更新最大窗口
    if(i - j + 1 > len) {
        len = i - j + 1;
        idx = j;
    }
}

最小窗口

// 窗口初始位置及长度
int idx = -1, len = INT_MAX;
for(int i = 0, j = 0; i < n; ++i) {
    // 将第 i 个元素加入窗口
    c[s[i] - 'A']++;
    
    // 新元素加入可能会导致窗口性质成立，但当前不一定是最小窗口
    while(check(c)) {
        // 更新最小窗口
        if(i - j + 1 < len) {
            len = i - j + 1;
            idx = j;
        }
        // 不断地从窗口左侧移出元素，直到性质不成立
        c[s[j++] - 'A']--;
    }
}

second version

最大窗口

template <class ForwardIterator, class Window>
std::pair<ForwardIterator, size_t> slide_window_max(ForwardIterator begin, ForwardIterator end, Window win) {
    auto idx = end;
    size_t len = 0, mx = 0;
    for(auto i = begin, j = begin; i != end; ++i) {
        // 将第 i 个元素添加到窗口里面
        win.add(*i); len++;
        
        // 添加新元素到窗口后可能会使窗口条件遭到破坏
        // 不断地从窗口左侧移出元素，直至窗口条件重新成立
        while(!win.check()) { win.remove(*j++); len--; }
        
        // 更新最大窗口
        if(len > mx) { mx = len; idx = j; }
    }
    return std::pair<ForwardIterator, size_t>(idx, mx);
}

template <class ForwardIterator, class Window>
std::pair<ForwardIterator, size_t> slide_window_min(ForwardIterator begin, ForwardIterator end, Window win) {
    auto idx = end;
    size_t len = 0, mn = INT_MAX;
    for(auto i = begin, j = begin; i != last; ++i) {
        // 将第 i 个元素加入窗口
        win.add(*i); len++;
        
        // 新元素加入可能会导致窗口性质成立，但当前不一定是最小窗口
        while(win.check()) {
            // 更新最小窗口
            if(len < mn) { mn = len; idx = j; }
            // 不断地从窗口左侧移出元素，直到性质不成立
            win.remove(*j++); len--;
        }
    }
    return std::pair<ForwardIterator, int>(idx, mn);
}

third version

template <class ForwardIterator, class Window, bool max_win = true>
std::tuple<ForwardIterator, ForwardIterator, auto> slide_window(ForwardIterator begin, ForwardIterator end, Window win) {
    auto best_begin = end, best_end = end;
    auto cur_value = win.value(), best_value = win.value();
    for(auto r = begin, p = begin; r != end; ++r) {
        // add element to win
        win.add(*r);

        while(win.check() != max_win) {
            // update res when scene is min window
            if constexpr(!max_win) if(win.value() < best_value) { best_begin = p; best_end = r; ++best_end; }
            // remove element from win
            win.remove(*p++);
        }
        // update res when scene is max window
        if constexpr(max_win) if(win.value() > best_value) { best_begin = p; best_end = r; ++best_end; }
    }
    return std::pair<ForwardIterator, size_t>(best_begin, best_end, best_value);
}
